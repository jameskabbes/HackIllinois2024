class WheelEncoder(Button):

    ticks: Ticks
    pin: int

    TICKS_PER_REVOLUTION: int = 192
    WHEEL_RADIUS: float = 25.0  # mm

    def __init__(self, config: Config):

        super().__init__(pin=config['pin'], pull_up=True)
        self.when_pressed = self.increment_ticks
        self.ticks = 0
